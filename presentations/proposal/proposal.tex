\begin{frame}{Introduction}
\framesubtitle{What Is the Minimum Wage?}
The minimum wage is a government-mandated wage floor designed to protect workers from excessively low pay.

\begin{figure}
  \centering

%------------------------------------------------
\begin{frame}{Introduction}
\framesubtitle{Occupational Dynamics and Wage Floors}
In Russia, minimum wage serves as a key labor policy instrument aimed at:
\begin{itemize}
  \item Ensuring basic income security
  \item Preventing in-work poverty
  \item Shaping employment conditions across sectors and regions
\end{itemize}

\textbf{Beyond social protection}, minimum wage policy can influence:
\begin{itemize}
  \item Employers' hiring and task allocation decisions
  \item Workers' occupational choices and sectoral mobility
  \item The distribution of employment across occupations
\end{itemize}
\end{frame}

\begin{frame}{Literature Review}
\framesubtitle{Existing Evidence}

\textbf{International Evidence:}
\begin{itemize}
\item Card \& Krueger (1994): Small employment effects in US
\item Neumark \& Wascher (2007): Negative effects in some contexts
\item Autor et al. (2016): Task-based vulnerability to labor shocks
\end{itemize}

\textbf{Russian Studies:}
\begin{itemize}
\item Lukiyanova (2010): MW compressed wage inequality
\item Gimpelson \& Kapeliushnikov (2011): Weak labor institutions
\item Regional MW studies limited to wage effects
\end{itemize}

\textbf{Gap:} No occupation-level analysis in Russian context
\end{frame}

\begin{frame}{Research Questions}
\framesubtitle{Primary Focus}

\textbf{Main Question:}

How do minimum wage changes affect occupational mobility and structure across Russian regions?

\vspace{0.3cm}

\textbf{Specific Questions:}
\begin{enumerate}
  \item Which occupations are most exposed to minimum wage increases?
  \item Do minimum wage changes induce occupational transitions across sectors and employment forms?
  \item How do occupational adjustment responses differ across regions?
\end{enumerate}
\end{frame}

%------------------------------------------------
\begin{frame}{Working Hypotheses}
\small
\framesubtitle{Proposed Testable Predictions}

\footnotesize{\textit{Note: These hypotheses are considered preliminary and may be refined during the analysis.}}

\begin{columns}[T]
\begin{column}{0.48\textwidth}

\textbf{H1: Occupational Exposure}
\begin{itemize}
  \item Low-skill and routine service occupations exhibit the highest exposure to minimum wage increases.
  \item Exposure intensity varies across regions.
\end{itemize}

\vspace{0.2cm}

\textbf{H2: Occupational Adjustment}
\begin{itemize}
  \item Minimum wage increases raise occupational transition rates among exposed workers.
  \item Transitions occur primarily within similar skill-level occupations.
\end{itemize}

\vspace{0.2cm}

\textbf{H4: Informal Sector Response}
\begin{itemize}
  \item Informal employment rises in high-MW regions for occupations heavily exposed to the minimum wage.
  \item Effects are stronger in regions with weaker labor institutions.
\end{itemize}

\end{column}

\begin{column}{0.48\textwidth}

\textbf{H3: Regional Heterogeneity}
\begin{itemize}
  \item High minimum-wage regions experience greater occupational displacement among exposed workers.
  \item Border regions exhibit higher cross-regional labor reallocation following minimum wage changes.
\end{itemize}

\vspace{0.2cm}

\textbf{H5: Sectoral Spillovers}
\begin{itemize}
  \item Minimum wage increases in one sector may affect wages and employment in adjacent sectors.
  \item Spillover magnitude depends on occupational similarity and regional labor market tightness.
\end{itemize}

\end{column}
\end{columns}
\end{frame}

\begin{frame}{Data Source}
\framesubtitle{Russian Longitudinal Monitoring Survey (RLMS)}

\begin{itemize}
\item Household panel: 1994-2024 (30 years)
\item 171,965 household-year observations
\item Nationally representative with regional identifiers
\item Occupational codes (3-digit OKZ)
\end{itemize}

\vspace{0.3cm}
\textbf{Sample Selection:}
\begin{itemize}
\item Working-age population (18-65)
\item Wage and salary workers
\item Complete occupation information
\end{itemize}

\textbf{Limitations:}
\begin{itemize}
\item Underrepresents informal sector
\item Occupational codes changed over time
\item Wage reporting may have measurement error
\end{itemize}
\end{frame}

\begin{frame}{Empirical Strategy}
\scriptsize
\framesubtitle{Three-Stage Approach}

\textbf{Stage 1: Descriptive Mapping}
\[
\text{MW Concentration}_{ort} = \frac{Emp_{ort, w \leq MW}}{Emp_{rt, w \leq MW}}, \quad w = \text{C4}
\]
Identify occupations most exposed to minimum wage increases.

\vspace{0.3cm}
\textbf{Stage 2: Difference-in-Differences (DiD)}
\[
Y_{irt} = \beta_0 + \beta_1 (\text{Post}_t \times \text{HighMW}_r) + \gamma_r + \delta_t + X_{irt}\theta + \epsilon_{irt}
\]
\begin{itemize}
\item \textbf{Assumptions:} parallel trends across regions, no spillovers beyond treatment
\item Regional (\(\gamma_r\)) and year (\(\delta_t\)) fixed effects
\item Controls: age, gender, education, sector (C2), firm type (C3), formal status (C8)
\item Dynamic effects: include leads/lags of MW exposure
\item Subsample analysis: formal vs informal, urban vs rural regions, high vs low MW regions
\end{itemize}

\vspace{0.3cm}
\textbf{Stage 3: Border Analysis}
\[
Y_i = \alpha + \beta \cdot \mathbf{1}\{\text{MW}_r > \text{MW}_{r'}\} + \epsilon_i
\]
\begin{itemize}
\item Exploit differences across adjacent regions to approximate quasi-experimental variation
\item Capture cross-regional occupational reallocation
\item Assumption: neighboring regions are comparable except for MW differences
\end{itemize}
\end{frame}

\begin{frame}{Methodology}
\scriptsize
\framesubtitle{Identification and Robustness}

\textbf{Identification Strategy:}
\begin{itemize}
\item Regional MW differences post-2007
\item Federal MW temporal changes
\item Interaction of region and time effects
\item Explicitly account for heterogeneity: occupation-level, sector-level, urban/rural
\end{itemize}

\vspace{0.3cm}
\textbf{Statistical Approach:}
\begin{itemize}
\item Regional and year fixed effects
\item Clustered standard errors at region-occupation level
\item Dynamic event study for temporal adjustment patterns
\item Pre-trend checks for DiD validity
\end{itemize}

\vspace{0.3cm}
\textbf{Robustness Checks:}
\begin{itemize}
\item Alternative wage measures (C4 vs G1\_1)
\item Different MW bite thresholds: 25\%, 30\%, 35\%
\item Placebo tests: false reform dates
\item Occupation fixed effects to control for systematic differences
\item Continuous treatment: MW relative to regional median wage
\item Subsample checks: formal vs informal sector, urban vs rural regions, high vs low MW regions
\item Link robustness to key assumptions (parallel trends, comparability of regions)
\end{itemize}
\end{frame}
 

%------------------------------------------------
\begin{frame}{Expected Findings}
\framesubtitle{Preliminary Expectations}

\textbf{Based on literature and descriptive patterns:}
\begin{itemize}
\item 5-8\% of workers directly affected by MW (higher in low-wage regions)
\item Routine occupations (retail, cleaning) most concentrated at MW
\item Public sector shows larger spillover effects due to wage grids
\item Border regions show occupational mobility toward lower-MW areas
\item Informal employment increases in high-MW regions for affected occupations
\end{itemize}

\vspace{0.3cm}
\textbf{Policy implication:} Targeted MW adjustments needed by occupation/region
\end{frame}
